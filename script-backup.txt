let runningTotal = 0;
let buffer = '0';
let previousOperator = null;
let isNumber = false;

const screen = document.querySelector(".screen");

function buttonClick(value) {
    if (isNaN(value)) {
        handleSymbol(value);
    } else {
        handleNumber(value);
    }
    screen.innerText = buffer;
}

function handleSymbol(value) {
    isNumber = false;
    switch(value) {
        case 'C':
            buffer = '0';
            runningTotal = 0;
            break;
        case '=':
            if(previousOperator == null) return;
            flushOperation(parseFloat(buffer));
            previousOperator = null;
            buffer = runningTotal;
            runningTotal = 0;
            break;
        case '←':
            isNumber = true;
            if(buffer.length === 1) {
                buffer = '0';
            } else {
                if(buffer == "Infinity" || buffer == "NaN") isNumber = false;
                buffer = buffer.slice(0, buffer.length - 1);
            }
            break;
        case '•':
            isNumber = true;
            if(buffer[buffer.length - 1] !== '.') buffer += '.';
            break;
        case '+':
            handleMath(value); break;
        case '−':
            handleMath(value); break;
        case '×':
            handleMath(value); break;
        case '÷':
            handleMath(value); break;
        case '^':
            handleMath(value); break;
        default:
            isNumber = true;
            break;
    }
}

function handleMath(symbol) {
    if(buffer == '0') {
        return;
    }

    const intBuffer = parseFloat(buffer);

    if(runningTotal === 0) {
        runningTotal = intBuffer;
    } else {
        flushOperation(intBuffer)
    }
    previousOperator = symbol;
    buffer = '0'
}

function flushOperation(intBuffer) {
    if(previousOperator === "+") {
        runningTotal += intBuffer;
    } else if(previousOperator === "−") {
        runningTotal -= intBuffer;
    } else if(previousOperator === "×") {
        runningTotal *= intBuffer;
    } else if(previousOperator === "^") {
        runningTotal **= intBuffer;
    } else {
        runningTotal /= intBuffer;
    }
    runningTotal = Math.round(runningTotal * 100) / 100;
}

function handleNumber(value) {
    if(buffer == '0' || !isNumber) {
        buffer = value;
    } else {
        buffer += value;
    }
    isNumber = true;
}

function init() {
    document.querySelector('.calc-buttons').addEventListener('click', (event) => {
        buttonClick(event.target.innerText);
    });
}

let runningTotal = 0;
let buffer = '0';
let previousOperator = null;
let isNumber = false;

const screen = document.querySelector(".screen");

function buttonClick(value) {
    if (isNaN(value)) {
        handleSymbol(value);
    } else {
        handleNumber(value);
    }
    screen.innerText = buffer;
}

function handleSymbol(value) {
    isNumber = false;
    switch(value) {
        case 'C':
        case 'c':
            buffer = '0';
            runningTotal = 0;
            break;
        case '=':
        case 'Enter':
            if(previousOperator == null) return;
            flushOperation(parseFloat(buffer));
            previousOperator = null;
            buffer = runningTotal;
            runningTotal = 0;
            break;
        case '←':
        case 'Backspace':
            isNumber = true;
            if(buffer.length === 1) {
                buffer = '0';
            } else {
                if(buffer == "Infinity" || buffer == "NaN") isNumber = false;
                buffer = buffer.slice(0, buffer.length - 1);
            }
            break;
        case '•':
        case '.':
            if(buffer != "Infinity" && buffer != "NaN") isNumber = true;
            if(buffer[buffer.length - 1] !== '.' && buffer != "Infinity" && buffer != "NaN") buffer += '.';
            break;
        case '+':
        case '−':
        case '-':
        case '×':
        case '*':
        case '÷':
        case '/':
        case '^':
            handleMath(value); break;
        default:
            isNumber = true;
            break;
    }
}

function handleMath(symbol) {
    if(buffer == '0') {
        return;
    }

    const intBuffer = parseFloat(buffer);

    if(runningTotal === 0) {
        runningTotal = intBuffer;
    } else {
        flushOperation(intBuffer)
    }
    previousOperator = symbol;
    buffer = '0'
}

function flushOperation(intBuffer) {
    if(previousOperator === "+") {
        runningTotal += intBuffer;
    } else if(previousOperator === "−" || previousOperator === "-") {
        runningTotal -= intBuffer;
    } else if(previousOperator === "×" || previousOperator === "*") {
        runningTotal *= intBuffer;
    } else if(previousOperator === "^") {
        runningTotal **= intBuffer;
    } else {
        runningTotal /= intBuffer;
    }
    runningTotal = Math.round(runningTotal * 1000000) / 1000000;
}

function handleNumber(value) {
    if(buffer == '0' || !isNumber) {
        buffer = value;
    } else {
        buffer += value;
    }
    isNumber = true;
}

function init() {
    document.querySelector('.calc-buttons').addEventListener('click', (event) => {
        buttonClick(event.target.innerText);
    });
    document.addEventListener('keydown', (e) => {
        buttonClick(e.key);
    });
}
